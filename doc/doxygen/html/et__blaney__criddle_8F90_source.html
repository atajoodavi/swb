9 !!****h* SWB/et_blaney_criddle
10 ! NAME
11 ! et_blaney_criddle.f95 - Evapotranspiration calculation using the
12 ! Blaney-Criddle method.
13 !
14 ! SYNOPSIS
15 ! This module calculates evapotranspiration using the Blaney-Criddle
16 ! method, as modified by Jensen and others (1990).
17 !
18 ! NOTES
19 ! Original method is documented in:
20 !
21 ! Blaney, H.F. and W.D. Criddle. 1950. Determining water requirements
22 ! in irrigated areas from climatological and irrigation data.
23 ! Soil Conservation Service Technical Paper 96. Soil Conservation Service,
24 ! U.S. Dept. of Agriculture: Washington, D.C.
25 !
26 ! Method included here has been modified for daily ET calculation and is
27 ! documented in:
28 !
29 ! Jensen, M.E, R.D Burman, and R.G. Allen. 1990. Evapotranspiration and
30 ! Irrigation Water Requirements. ASCE Manuals and Reports of Engineering
31 ! Practice No. 70. New York: ASCE.
32 !
33 !!***

67 subroutine et_bc_initialize( grd, sFileName )
68  !! Preconfigures necessary information from the time-series file 'sFileName'

87 subroutine et_bc_computeet( pGrd, iDayOfYear, rRH, &
88  rminrh, rwindspd, rsunpct )
89  !! Computes the potential ET for each cell, based on the meteorological
90  !! data given. Stores cell-by-cell PET values in the model grid.
91  !! Note: for the T-M model, it's constant scross the grid
92  !!
93  ! [ ARGUMENTS ]
94  type ( t_general_grid ),pointer :: pgrd
95  integer (kind=c_int),intent(in) :: idayofyear
96  real (kind=c_float),intent(in) :: rrh,rminrh,rwindspd,rsunpct
97  ! [ LOCALS ]
98  real (kind=c_float) :: rdelta,rt,rsunfrac,rpratio,rabc,rbbc,rf
99  integer (kind=c_int) :: icol,irow
100  ! [ CONSTANTS ]
101  real (kind=c_float),parameter :: unit_conv = 0.313_c_float / 25.4_c_float
102 
103  call assert( LOGICAL(rSunPct>=rZERO,kind=c_bool), "Missing data for percent sunshine" )
104  call assert( LOGICAL(rMinRH>=rZERO,kind=c_bool),"Missing data for relative humidity" )
105 
106 ! rDelta = 0.4093_c_float * sin( (dpTWOPI * iDayOfYear / 365.0_c_float) - 1.405_c_float )
107  rdelta = solar_declination(idayofyear, 365)
108 
109  rpratio = ( 24.0_c_float * acos(-tan(rlatitude) * tan(rdelta)) / dppi ) / rsunannual
110 
111  rsunfrac = rsunpct / rhundred
112  rabc = ( 0.0043_c_float * rminrh) - rsunfrac - 1.41_c_float
113 
114 !
115 ! The following is what was originally coded by WES and Vic. Origin unclear.
116 ! Possibly (probably) the source is Doorenbos and Pruitt (1977)?
117 ! More like likely source is Allen and Pruitt (1986) "Rational Use of the FAO BC Formula"
118 !
119  rbbc = 0.81917_c_float - 0.0040922_c_float*rminrh +1.0705_c_float*rsunfrac + &
120  0.065649*rwindspd - 0.0059684*rminrh*rsunfrac - &
121  0.0005967*rminrh*rwindspd
122 
123 ! Formulation for rBbc as given below is from Jensen et al (1990):
124 
125 ! rBbc = 0.908 - (0.00483*rMinRH) + (0.7949*rSunFrac) &
126 ! + (0.768*(LOG(rWindSpd+1.)**2)) &
127 ! - (0.0038*rMinRH*rSunFrac) &
128 ! - (0.000443*rMinRH*rWindSpd) &
129 ! + (0.281*LOG(rSunFrac+1.)) &
130 ! - (0.00975*LOG(rWindSpd+1.)*(LOG(rMinRH+1.)**2)*LOG(rSunFrac+1.))
131 
132 
133  do irow=1,pgrd%iNY
134  do icol=1,pgrd%iNX
135 
136  rt = ftoc(pgrd%Cells(icol,irow)%rTAvg)
137 
138  rf = rpratio * ( 8.13_c_float + 0.46_c_float * rt)
139 
140  if ( pgrd%Cells(icol,irow)%rTAvg <= rfreezing ) then
141  pgrd%Cells(icol,irow)%rReferenceET0 = rzero
142  else
143  pgrd%Cells(icol,irow)%rReferenceET0 = (rabc + rf * rbbc) / rmm_per_inch
144  write(unit=lu_log,fmt=*)'TEST',idayofyear, "rMinRH: ", rminrh, &
145  "Sunfrac: ", rsunfrac, "rPratio: ",rpratio, &
146  "rAbc: ", rabc, "rF: ", rf, "rBbc: ", rbbc, "ET0: ", pgrd%Cells(icol,irow)%rReferenceET0
147  end if
148 
149  end do
150 
151  end do
152 
153  return
154 end subroutine et_bc_computeet
