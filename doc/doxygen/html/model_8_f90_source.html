03035                                                    ! model options, flags, and other settings

03074           if ( cel%lDownhillMarked ) cycle

03076           iUpstreamCount = 0

03108             .or. (iUpstreamCount == 1 .and. lCircular)) then

03157         end where
03158 
03159 !#ifdef GRAPHICS_SUPPORT
03160 !        call genericgraph(pTempGrid)
03161 !#endif
03162 
03163 #ifdef DEBUG_PRINT
03164 
03165       call grid_WriteArcGrid("iteration"//TRIM(int2char(iPasses))// &

03168           REAL(pTempGrid%iData,kind=T_SGL))
03169 
03170 #endif
03171 
03172       else
03173         ! reset iteration counter
03174         iNumIterationsNochange = 0
03175       end if
03176 
03177       if(iOrderCount == iNumGridCells) exit
03178       iPasses = iPasses+1
03179       write(UNIT=LU_LOG,FMT=*) 'Iteration ',iPasses,'  ',iOrderCount,&
03180          ' of ',iNumGridCells,' cells have been configured'
03181     end do
03182 
03183     write(UNIT=LU_LOG,FMT=*) "  Number of passes required: ",iPasses
03184     write(UNIT=LU_LOG,FMT=*) ""
03185     flush(UNIT=LU_LOG)
03186 
03187     open(UNIT=LU_ROUTING, FILE='swb_routing.bin',FORM='UNFORMATTED', &
03188       status='REPLACE',ACCESS='STREAM')
03189     write(LU_ROUTING) iOrderCount
03190 
03191     do i=1,iOrderCount
03192       write(LU_ROUTING) iOrderJ(i),iOrderI(i)
03193     end do
03194     flush(UNIT=LU_ROUTING)
03195     close(UNIT=LU_ROUTING)
03196 
03197   else ! routing table already exists
03198 
03199     pGrd%Cells%lDownhillMarked = lTRUE
03200     open(UNIT=LU_ROUTING, FILE='swb_routing.bin',FORM='UNFORMATTED', &
03201       ACCESS='STREAM')

03218     write(UNIT=LU_LOG,FMT=*)  "*****************************************************************************"
