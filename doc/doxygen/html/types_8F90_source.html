2389 ! Communications of the ACM, Vol. 11, No. 10 (October, 1968).
2390 ! Modified from code found at:
2391 ! http://aa.usno.navy.mil/faq/docs/JD_Formula.html

2405 subroutine gregorian_date(iJD, iYear, iMonth, iDay, iOrigin)
2406 
2407  ! [ ARGUMENTS ]
2408  integer (kind=T_INT) :: ijd
2409  integer (kind=T_INT), intent(inout) :: iyear, imonth, iday
2410  integer (kind=T_INT), optional :: iorigin
2411  ! [ LOCALS ]
2412  integer (kind=T_INT) ii,ij,ik,il,in
2413  integer (kind=T_INT) :: ioffset
2414 
2415  if(present(iorigin)) then
2416  ioffset = iorigin
2417  else
2418  ioffset = 0
2419  endif
2420 
2421  ! allow for an alternate "origin" to be specified... technically,
2422  ! this is no longer a "Julian" day, but alas... This modification
2423  ! was required in order to process the "time" variables from global
2424  ! climate models, which seem to be defined as something like this:
2425  ! time:units = "days since 1960-01-01 00:00:00"
2426  !
2427  ! for the above example, JD = 2436935 on the first day; the NetCDF "time"
2428  ! variable will be equal to 0. Thus, in order to get the conversion
2429  ! right, we must add 0 + 2436935 to yield a true Julian Day.
2430 
2431  ijd = ijd + ioffset
2432 
2433  il= ijd + 68569_t_int
2434  in= 4*il / 146097_t_int
2435  il= il - (146097_t_int * in + 3_t_int)/4_t_int
2436  ii= 4000_t_int * (il + 1_t_int) / 1461001_t_int
2437  il= il - 1461_t_int * ii / 4_t_int + 31_t_int
