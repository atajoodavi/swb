69 subroutine sm_thornthwaite_mather_initialize ( pGrd, pConfig )
70  !! Preconfigures soil moisture for the model grid 'pGrd'.
71  ! [ ARGUMENTS ]
72  type ( t_general_grid ),pointer :: pgrd
73  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
74  ! model options, flags, and other settings
75  ! [ LOCALS ]
76  integer (kind=T_INT) :: irow,icol
77  type ( t_cell ),pointer :: cel
78 
79  ! Initialize the accumulated water loss for each cell according to the
80  ! initial soil moisture.
81 
82  do irow=1,pgrd%iNY
83  do icol=1,pgrd%iNX
84 
85  cel => pgrd%Cells(icol,irow)
86 
87  if ( cel%rSoilWaterCap > rnear_zero ) then
88 
89  !! Constrain rSoilWaterCap to limits of Thornthwaite-Mather tables
90 
91  if ( cel%rSoilWaterCap < 0.5_t_sgl ) then
92  write(unit=lu_log,fmt="(a,f10.3,a,i4,',',i4,a)") &
93  'Soil water capacity', cel%rSoilWaterCap,' for cell ',irow,icol, &
94  ' out of range; adjusted to 0.5'
95  flush(unit=lu_log)
96  cel%rSoilWaterCap = 0.51_t_sgl
97  else if ( cel%rSoilWaterCap > 17.5_t_sgl ) then
98  write(unit=lu_log,fmt="(a,f10.3,a,i4,',',i4,a)") &
99  'Soil water capacity', cel%rSoilWaterCap,' for cell ',irow,icol, &
100  ' out of range; adjusted to 17.5'
101  flush(unit=lu_log)
102  cel%rSoilWaterCap = 17.49_t_sgl
103  end if
104 
105  !! convert input soil moisture (as percent of soil water capacity)
106  !! TO soil moisture in inches
107  cel%rSoilMoisture = (cel%rSoilMoisturePct/rhundred) * cel%rSoilWaterCap
108 
109  !! back-calculate initial accumulated potential water loss term
110  !! given initial soil moisture
111 
112 #ifdef TM_TABLE
113 
114  ! calculate APWL from T-M table
115  cel%rSM_AccumPotentWatLoss = &
116  grid_searchcolumn(pgrd=gwlt, &
117  rxval=cel%rSoilWaterCap, &
118  rzval=cel%rSoilMoisture, &
119  rnodata=-rone)
120 
121 #else
122 
123  ! calculate APWL from equation
124  cel%rSM_AccumPotentWatLoss = &
125  sm_thornthwaite_mather_apwl(cel%rSoilWaterCap,real(cel%rSoilMoisture, kind=T_DBL) )
126 
127 #endif
128 
129  end if
130  end do
131  end do
132 
133 #ifdef DEBUG_PRINT
134  call grid_writearcgrid("initial_APWL.asc",dpzero,dpone,dpzero, &
135  dpone,pgrd%Cells(:,:)%rSM_AccumPotentWatLoss)
136 #endif
137 
138  !! initialize Thorntwaite-Mather soil moisture accounting variables
139 
140  pgrd%Cells%rReferenceET0 = rzero
141 
142 end subroutine sm_thornthwaite_mather_initialize

147 function sm_thornthwaite_mather_soil_storage(rSWC, rAPWL) result(dpValue)
148 
149  real (kind=T_SGL), intent(in) :: rswc ! max soil-water capacity (inches)
150  real (kind=T_SGL), intent(in) :: rapwl ! accum pot. water loss (inches)
151 
152  real (kind=T_DBL) :: dpvalue
153 
154  ! equation as implemented in R;
155  ! sm.df$y = maximum soil-water capacity
156  ! sm.df$x = APWL
157  ! 10^(log10(sm.df$y) - (s.opt[[1]]*sm.df$y^e.opt[[1]]) * sm.df$x)
158 
159  dpvalue = dpzero
160 
161  if(rswc > rzero ) &
162 
163  dpvalue = 10_t_dbl**( log10(REAL(rswc,kind=t_dbl)) - &
164  ( abs(REAL(rapwl,kind=t_dbl)) * rtm_slope_term &
165  * real(rswc, kind=t_dbl)**rtm_exp_term ) )
166 
167 
168 end function sm_thornthwaite_mather_soil_storage

172 function sm_thornthwaite_mather_apwl(rSWC, dpSoilStorage) result(dpValue)
173 
174  real (kind=T_SGL), intent(in) :: rswc ! max soil-water capacity (inches)
175  real (kind=T_DBL), intent(in) :: dpsoilstorage ! curr soil storage (inches)
176 
177  real (kind=T_DBL) :: dpvalue
178 
179  ! equation as implemented in R;
180  ! sm.df$y = maximum soil-water capacity
181  ! sm.df$x = APWL
182  ! (log10(sm.df$y) - log10(sm.df$pred)) / (s.opt[[1]] * sm.df$y^e.opt[[1]])
183 
184  dpvalue = dpzero
185 
186  if(rswc > rzero .and. dpsoilstorage > dpzero) &
187 
188  dpvalue = -( log10(REAL(rswc,kind=t_dbl)) - log10(dpsoilstorage)) / &
189  ( rtm_slope_term * REAL(rswc,kind=t_dbl)**rtm_exp_term )
190 
191 end function sm_thornthwaite_mather_apwl
