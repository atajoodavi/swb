128 subroutine runoff_updatecurvenumber(pConfig, cel,iJulDay)
129  !! Updates the curve numbers for this iteration
130  ! [ ARGUMENTS ]
131  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that con
132  ! model options, flags, and other se
133  type (t_cell),pointer :: cel
134  integer (kind=c_int),intent(in) :: ijulday
135  ! [ LOCALS ]
136  real (kind=c_float) :: rtotalinflow
137  real (kind=c_float) :: rtempcn
138  real (kind=c_float) :: rpf
139 
140  !! update curve numbers based on antecedent moisture conditions
141 ! real (kind=c_float),dimension(5),parameter :: rDRY_COEFS = (/ &
142 ! 1.44206581732462E-06_c_float, &
143 ! -2.54340415305462E-04_c_float, &
144 ! 2.07018739405394E-02_c_float, &
145 ! -7.67877072822852E-03_c_float, &
146 ! 2.09678222732103_c_float /)
147 ! real (kind=c_float),dimension(5),parameter :: rWET_COEFS = (/ &
148 ! -6.20352282661163E-07_c_float, &
149 ! 1.60650096926368E-04_c_float, &
150 ! -2.03362629006156E-02_c_float, &
151 ! 2.01054923513527_c_float, &
152 ! 3.65427885962651_c_float /)
153 
154  ! Here goes...
155  rtotalinflow = sum(cel%rNetInflowBuf)
156 
157  ! Correct the curve number...
158  if(cel%rCFGI>pconfig%rLL_CFGI &
159  .and. cel%rSoilWaterCap > rnear_zero) then
160 ! cel%rAdjCN = MIN(98_c_float,cel%rAdjCN + &
161 ! ((98_c_float-cel%rAdjCN) * cel%rSoilMoisture / cel%rSoilWaterCap))
162 
163  rpf = prob_runoff_enhancement(cel%rCFGI,pconfig%rLL_CFGI,pconfig%rUL_CFGI)
164 
165  ! use probability of runoff enhancement to calculate a weighted
166  ! average of curve number under Type II vs Type III antecedent
167  ! runoff conditions
168  cel%rAdjCN = cel%rBaseCN * (1-rpf) + &
169  (cel%rBaseCN / (0.427 + 0.00573 * cel%rBaseCN) * rpf)
170 
171  else if ( lf_model_growingseason(pconfig, ijulday) ) then
172 
173  if ( rtotalinflow < pconfig%rDRY_GROWING ) then ! AMC I
174 
175 
176 ! cel%rAdjCN = polynomial(cel%rBaseCN,rDRY_COEFS)
177 
178 ! The following comes from page 192, eq. 3.145 of "SCS Curve Number
179 ! Methodology"
180 
181  cel%rAdjCN = cel%rBaseCN / (2.281 - 0.01281 * cel%rBaseCN)
182 
183  else if ( rtotalinflow >= pconfig%rDRY_GROWING &
184  .and. rtotalinflow < pconfig%rWET_GROWING ) then ! AMC II
185 
186  cel%rAdjCN = real(cel%rbasecn)
187 
188  else ! AMC III
189 
190 ! cel%rAdjCN = polynomial(cel%rBaseCN,rWET_COEFS)
191  cel%rAdjCN = cel%rBaseCN / (0.427 + 0.00573 * cel%rBaseCN)
192  end if
193 
194  else ! dormant (non-growing) season
195 
196  if ( rtotalinflow < pconfig%rDRY_DORMANT ) then ! AMC I
197 
198 ! cel%rAdjCN = polynomial(cel%rBaseCN,rDRY_COEFS)
199  cel%rAdjCN = cel%rBaseCN / (2.281 - 0.01281 * cel%rBaseCN)
200 
201  else if ( rtotalinflow >= pconfig%rDRY_DORMANT &
202  .and. rtotalinflow < pconfig%rWET_DORMANT ) then ! AMC II
203 
204  cel%rAdjCN = real(cel%rbasecn)
205 
206  else ! AMC III
207 
208 ! cel%rAdjCN = polynomial(cel%rBaseCN,rWET_COEFS)
209  cel%rAdjCN = cel%rBaseCN / (0.427 + 0.00573 * cel%rBaseCN)
210 
211  end if
212 
213  end if
214 
215  ! ensure that whatever modification have been made to the curve number
216  ! remain within reasonable bounds
217  cel%rAdjCN = min(cel%rAdjCN,rhundred)
218  cel%rAdjCN = max(cel%rAdjCN,rzero)
219

224 function runoff_cellrunoff_curvenumber(pConfig, cel, iJulDay) result(rOutFlow)
225  !! Calculates a single cell's runoff using curve numbers
226  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
227  ! model options, flags, and other settings
228  type (t_cell),pointer :: cel
229  integer (kind=c_int),intent(in) :: ijulday
230  ! [ RETURN VALUE ]
231  real (kind=c_float) :: routflow
232  ! [ LOCALS ]
233  real (kind=c_float) :: rp
234  real (kind=c_float) :: rcn_05
235  real (kind=c_float) :: rsmax
236 
237  rp = cel%rNetRainfall &
238  + cel%rSnowMelt &
239  + cel%rIrrigationAmount &
240  + cel%rInFlow
241 
242  call runoff_updatecurvenumber(pconfig,cel,ijulday)
243 
244  rsmax = (rthousand / cel%rAdjCN) - rten
245 
246  if(pconfig%iConfigureInitialAbstraction == &
247  config_sm_init_abstraction_tr55) then
248 
249  if ( rp > rpoint2*rsmax ) then
250  routflow = ( rp - rpoint2*rsmax )**2 / (rp + rpoint8*rsmax)
251  else
252  routflow = rzero
253  end if
254 
255  else if(pconfig%iConfigureInitialAbstraction == &
256  config_sm_init_abstraction_hawkins) then
257 
258  ! Equation 9, Hawkins and others, 2002
259  rcn_05 = 100_c_float / &
260  ((1.879_c_float * ((100_c_float / cel%rAdjCN) - 1_c_float )**1.15_c_float) +1_c_float)
261 
262 
263  ! Equation 8, Hawkins and others, 2002
264  rsmax = 1.33_c_float * ( rsmax ) ** 1.15_c_float
265 
266  ! now consider runoff if Ia ~ 0.05S
267  if ( rp > 0.05_c_float * rsmax ) then
268  routflow = ( rp - 0.05_c_float * rsmax )**2 / (rp + 0.95_c_float*rsmax)
269  else
270  routflow = rzero
271  end if
272 
273  else
274  call assert(lfalse, "Illegal initial abstraction method specified" )
275  end if
276 
277  return
278 
279 end function runoff_cellrunoff_curvenumber
