8 module et_blaney_criddle
9 !!****h* SWB/et_blaney_criddle
10 ! NAME
11 ! et_blaney_criddle.f95 - Evapotranspiration calculation using the
12 ! Blaney-Criddle method.
13 !
14 ! SYNOPSIS
15 ! This module calculates evapotranspiration using the Blaney-Criddle
16 ! method, as modified by Jensen and others (1990).
17 !
18 ! NOTES
19 ! Original method is documented in:
20 !
21 ! Blaney, H.F. and W.D. Criddle. 1950. Determining water requirements
22 ! in irrigated areas from climatological and irrigation data.
23 ! Soil Conservation Service Technical Paper 96. Soil Conservation Service,
24 ! U.S. Dept. of Agriculture: Washington, D.C.
25 !
26 ! Method included here has been modified for daily ET calculation and is
27 ! documented in:
28 !
29 ! Jensen, M.E, R.D Burman, and R.G. Allen. 1990. Evapotranspiration and
30 ! Irrigation Water Requirements. ASCE Manuals and Reports of Engineering
31 ! Practice No. 70. New York: ASCE.
32 !
33 !!***

47 subroutine et_bc_configure( sRecord )
48  !! Configures the module, using the command line in 'sRecord'
49  ! [ ARGUMENTS ]
50  character (len=*),intent(inout) :: srecord
51  ! [ LOCALS ]
52  character (len=256) :: soption
53  integer (kind=T_INT) :: istat
54 
55  write(unit=lu_log,fmt=*) "Configuring Blaney-Criddle PET model"
56 
57  call chomp( srecord,soption )
58  read ( unit=soption, fmt=*, iostat=istat ) rlatitude
59  call assert( LOGICAL(iStat==0,kind=T_LOGICAL), "Could not read the latitude" )
60  rlatitude = dptwopi * rlatitude / 360.0_t_sgl
61 
62  return
63 end subroutine et_bc_configure

65 subroutine et_bc_initialize( grd, sFileName )
66  !! Preconfigures necessary information from the time-series file 'sFileName'
67  !! and based on the model grid 'grd'.
68  ! [ ARGUMENTS ]
69  type ( t_general_grid ),pointer :: grd
70  character (len=*),intent(in) :: sfilename
71  ! [ LOCALS ]
72  real (kind=T_SGL) :: rdelta
73  integer (kind=T_INT) :: iday
74 
75  write(unit=lu_log,fmt=*)"Initializing Blaney-Criddle PET model"
76  do iday=1,365
77  rdelta = 0.4093_t_sgl * sin( (dptwopi * iday / 365.0_t_sgl) - 1.405_t_sgl )
78  rsunannual = rsunannual + 24.0_t_sgl * acos(-tan(rlatitude) * tan(rdelta)) / dppi
79  end do
80 
81  return

86 subroutine et_bc_computeet( pGrd, iDayOfYear, rRH, &
87  rminrh, rwindspd, rsunpct )
88  !! Computes the potential ET for each cell, based on the meteorological
89  !! data given. Stores cell-by-cell PET values in the model grid.
90  !! Note: for the T-M model, it's constant scross the grid
91  !!
92  ! [ ARGUMENTS ]
93  type ( t_general_grid ),pointer :: pgrd
94  integer (kind=T_INT),intent(in) :: idayofyear
95  real (kind=T_SGL),intent(in) :: rrh,rminrh,rwindspd,rsunpct
96  ! [ LOCALS ]
97  real (kind=T_SGL) :: rdelta,rt,rsunfrac,rpratio,rabc,rbbc,rf
98  integer (kind=T_INT) :: icol,irow
99  ! [ CONSTANTS ]
100  real (kind=T_SGL),parameter :: unit_conv = 0.313_t_sgl / 25.4_t_sgl
101 
102  call assert( LOGICAL(rSunPct>=rZERO,kind=T_LOGICAL), "Missing data for percent sunshine" )
103  call assert( LOGICAL(rMinRH>=rZERO,kind=T_LOGICAL),"Missing data for relative humidity" )
104 
105  rdelta = 0.4093_t_sgl * sin( (dptwopi * idayofyear / 365.0_t_sgl) - 1.405_t_sgl )
106  rpratio = ( 24.0_t_sgl * acos(-tan(rlatitude) * tan(rdelta)) / dppi ) / rsunannual
107 
108  rsunfrac = rsunpct / rhundred
109  rabc = ( 0.0043_t_sgl * rminrh) - rsunfrac - 1.41_t_sgl
110 
111 !
112 ! The following is what was originally coded by WES and Vic. Origin unclear.
113 ! Possibly (probably) the source is Doorenbos and Pruitt (1977)?
114 !
115 ! rBbc = 0.81917_T_SGL - 0.0040922_T_SGL*rMinRH +1.0705_T_SGL*rSunFrac + &
116 ! 0.065649*rWindSpd - 0.0059684*rMinRH*rSunFrac - &
117 ! 0.0005967*rMinRH*rWindSpd
118 
119 ! Formulation for rBbc as given below is from Jensen et al (1990):
120 
121  rbbc = 0.908 - (0.00483*rminrh) + (0.7949*rsunfrac) &
122  + (0.768*(log(rwindspd+1.)**2)) &
123  - (0.0038*rminrh*rsunfrac) &
124  - (0.000443*rminrh*rwindspd) &
125  + (0.281*log(rsunfrac+1.)) &
126  - (0.00975*log(rwindspd+1.)*(log(rminrh+1.)**2)*log(rsunfrac+1.))
127 
128 
129  do irow=1,pgrd%iNY
130  do icol=1,pgrd%iNX
131 
132  rt = ftoc(pgrd%Cells(icol,irow)%rTAvg)
133 
134  rf = rpratio * ( 8.13_t_sgl + 0.46_t_sgl * rt)
135 
136  if ( pgrd%Cells(icol,irow)%rTAvg <= rfreezing ) then
137  pgrd%Cells(icol,irow)%rSM_PotentialET = rzero
138  else
139  !write(UNIT=LU_LOG,FMT=*)'TEST',iDayOfYear,rPratio,rAbc,rBbc
140  pgrd%Cells(icol,irow)%rSM_PotentialET = (rabc + rf * rbbc) / rmm_per_inch
141  end if
142 
143  end do
144 
145  end do
146 
147  return
148 end subroutine et_bc_computeet
149
