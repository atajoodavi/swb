9 !!****h* SWB/et_thornthwaite_mather
10 ! NAME
11 ! et_thornthwaite_mather.f95 - Evapotranspiration calculation using the
12 ! Thornthwaite-Mather method.
13 !
14 ! SYNOPSIS
15 ! This module calculates evapotranspiration using the Thornthwaite-Mather
16 ! method.
17 !
18 ! NOTES
19 ! Original method is documented in:
20 !
21 ! Thornthwaite, C.W., and Mather, J.R., 1955, The water balance,
22 ! in Publications in climatology, v. 8, no. 1: Centerton, N.J.,
23 ! Laboratory of Climatology, p. 1-104.
24 !
25 ! Thornthwaite, C.W., and Mather, J.R., 1957, Instructions and tables
26 ! for computing potential evapotranspiration and the water balance,
27 ! in Publications in climatology ; v. 10, no. 3: Centerton, N.J.,
28 ! Laboratory of Climatology, p. 185-311.
29 !
30 !!***

71 subroutine et_tm_initialize( pGrd, pConfig, sFileName )
72  !! Preconfigures necessary information from the time-series file 'sFileName'
73  !! and based on the model grid 'pGrd'.
74  ! [ ARGUMENTS ]
75  type ( t_general_grid ),pointer :: pgrd
76  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
77  ! model options, flags, and other settings
78  character (len=*),intent(in) :: sfilename
79  ! [ LOCALS ]
80  integer (kind=c_int) :: ijulianday, istat, imonth, iday, iyear,i
81  integer (kind=c_int) :: idayofyear
82  real(kind=c_float) :: rprecip, ravgt, rmint, rmaxt, rrh, rminrh, rwindspd, rsunpct
83  real(kind=c_double) :: rmonthavg,rmonthindex
84  character (len=256) :: sbuf
85  character (len=3) :: smonthname
86  logical (kind=c_bool) :: lmonthend
87  logical (kind=c_bool) :: lopened
88  integer (kind=c_int) :: ilu
89  integer (kind=c_int), dimension(12) :: imonthlycount
90  real (kind=c_double), dimension(12) :: rmonthlysum
91  ! [ LOCAL CONSTANTS ]
92 
94  real (kind=c_double),parameter :: rexp=1.514_c_double
95  write(unit=lu_log,fmt=*)"Initializing Thornthwaite-Mather ET model with annual data ", trim(sfilename)
96 
97 
101 
102  open ( lu_temp, file=trim(sfilename), iostat=istat )
103  call assert( istat == 0, "Could not open time series file " // trim(sfilename), &
104  trim(__file__),__line__)
105 
106  ! Now, compute the monthly-average temperatures for all months
107  imonthlycount = 0
108  rmonthlysum = rzero
109 
110  do
111  read ( unit=lu_temp, fmt="(a256)", iostat=istat ) sbuf
112  if ( istat<0 ) exit ! Here on EOF
113  call assert( istat == 0, "Cannot read record from time-series file" )
114  if ( sbuf(1:1) == '#' ) cycle ! Check comments
115  call cleanupcsv( sbuf )
116 
117  read ( unit=sbuf, fmt=*, iostat=istat ) imonth, iday, iyear, ravgt, &
118  rprecip, rrh, rmaxt, rmint, &
119  rwindspd, rminrh, rsunpct
120 
121  if (istat/=0) then
122  write(unit=lu_log,fmt=*)" Skipping: ",trim(sbuf)
123  cycle
124  end if
125 
126  call lookupmonth(imonth, iday, iyear,idayofyear, &
127  smonthname,lmonthend)
128 
129  rmonthlysum(imonth) = rmonthlysum(imonth) + real(ravgt, kind=c_double)
130  imonthlycount(imonth) = imonthlycount(imonth) + 1
131  end do
132 
133  ! Now, compute monthly temperature indices (leave out the no-data months)
134  ! And sum to get the annual index
135 
136  rannualindex = rzero
137  do i=1,12
138  if ( imonthlycount(i) > 0 ) then
139  rmonthavg = rmonthlysum(i) / real(iMonthlyCount(i), kind=c_double)
140  if ( rmonthavg > dpfreezing ) then
142  rmonthindex = (ftoc(rmonthavg) / 5_c_double) ** 1.514_c_double
143  rannualindex = rannualindex + rmonthindex
144  end if
145  end if
146  end do
147 
149  rexponenta = 6.75e-7 * rannualindex**3 &
150  - 7.71e-5 * rannualindex**2 &
151  + 1.7921e-2 * rannualindex &
152  + 0.49239
153 
154  close (unit=lu_temp)
155 
156  write(unit=lu_log,fmt=*) " Thornthwaite-Mather annual heat index = ",rannualindex
157  write(unit=lu_log,fmt=*) " Thornthwaite-Mather exponent 'a' = ",rexponenta
158 
159 end subroutine et_tm_initialize

161 subroutine et_tm_computeet( pGrd, pConfig, iDayOfYear)
162  !! Computes the potential ET for each cell, based on the meteorological
163  !! data given. Stores cell-by-cell PET values in the model grid.
164  !! Note: for the T-M model, it's constant scross the grid
165  !!
166  ! [ ARGUMENTS ]
167  type ( t_general_grid ),pointer :: pgrd
168  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
169  ! model options, flags, and other settings
170  integer (kind=c_int),intent(in) :: idayofyear
171  real (kind=c_double) :: rpotet, rdecl, romega_s, rn, rpotet_adj
172 
173  ! [ locals ]
174  integer (kind=c_int) :: icol, irow
175 ! real (kind=c_float),dimension(3),parameter :: rHighTPoly = &
176 ! (/ -5.25625072726565D-03, 1.04170341298537D+00, - 44.3259754866234D+00 /)
177  type (t_cell), pointer :: cel
178  real (kind=c_double) :: rtempc
179 
180 
181  rdecl = 0.4093_c_double * sin((dptwopi * real(idayofyear, kind=c_double)/365.0_c_double) - 1.405_c_double )
182 ! rDecl = solar_declination(iDayOfYear, MODEL_SIM%daysperyear())
183  romega_s = sunset_angle(real(rLatitude * dpPI_OVER_180, kind=c_double), rdecl)
184  rn = daylight_hours(romega_s)
185 
186  do irow=1,pgrd%iNY
187  do icol=1,pgrd%iNX ! last subscript in a Fortran array should be the slowest changing
188  cel => pgrd%Cells(icol,irow)
189 
190  rtempc = ftoc(real(cel%rtavg, kind=c_double))
191 
192  if (pgrd%Cells(icol,irow)%rTAvg <= rfreezing ) then
193  rpotet = dpzero
194  else if ( cel%rTAvg <= 79.7_c_float ) then
195 
198  rpotet = (16_c_double * (10_c_double*rtempc / rannualindex)**rexponenta) &
199  / 25.4_c_double / 30_c_double
200 
201  else
202 
206  rpotet = (-415.85_c_double + 32.24_c_double*rtempc -0.43_c_double*rtempc**2) &
207  / 25.4_c_double &
208  / 30_c_double
209 
210  end if
211 
214  rpotet_adj = rpotet * rn / 12.0_c_double
215 
216  pgrd%Cells(icol,irow)%rReferenceET0 = rpotet_adj
217 
218  end do
219 
220  end do
221 
222 end subroutine et_tm_computeet
