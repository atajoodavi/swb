8 module et_thornthwaite_mather
9 !!****h* SWB/et_thornthwaite_mather
10 ! NAME
11 ! et_thornthwaite_mather.f95 - Evapotranspiration calculation using the
12 ! Thornthwaite-Mather method.
13 !
14 ! SYNOPSIS
15 ! This module calculates evapotranspiration using the Thornthwaite-Mather
16 ! method.
17 !
18 ! NOTES
19 ! Original method is documented in:
20 !
21 ! Thornthwaite, C.W., and Mather, J.R., 1955, The water balance,
22 ! in Publications in climatology, v. 8, no. 1: Centerton, N.J.,
23 ! Laboratory of Climatology, p. 1-104.
24 !
25 ! Thornthwaite, C.W., and Mather, J.R., 1957, Instructions and tables
26 ! for computing potential evapotranspiration and the water balance,
27 ! in Publications in climatology ; v. 10, no. 3: Centerton, N.J.,
28 ! Laboratory of Climatology, p. 185-311.
29 !
30 !!***

49 subroutine et_tm_configure( sRecord )
50  !! Configures the module, using the command line in 'sRecord'
51  ! [ ARGUMENTS ]
52  character (len=*),intent(inout) :: srecord
53  ! [ LOCALS ]
54  character (len=256) :: soption
55  integer (kind=T_INT) :: istat
56 
57  write(unit=lu_log,fmt=*) "Configuring Thornthwaite-Mather ET model"
58  call chomp( srecord,soption )
59  read ( unit=soption, fmt=*, iostat=istat ) rlatitude
60  call assert( istat == 0, &
61  "Could not read the latitude" )
62  rtanlatitude = tan(dptwopi * rlatitude / 360.0_t_sgl )
63 ! write(UNIT=LU_LOG,FMT=*)" ---> ",rLatitude, rTanLatitude
64 
65  return

68 subroutine et_tm_initialize( pGrd, pConfig, sFileName )
69  !! Preconfigures necessary information from the time-series file 'sFileName'
70  !! and based on the model grid 'pGrd'.
71  ! [ ARGUMENTS ]
72  type ( t_general_grid ),pointer :: pgrd
73  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
74  ! model options, flags, and other settings
75  character (len=*),intent(in) :: sfilename
76  ! [ LOCALS ]
77  integer (kind=T_INT) :: ijulianday, istat, imonth, iday, iyear,i
78  integer (kind=T_INT) :: idayofyear
79  real(kind=T_SGL) :: rprecip, ravgt, rmint, rmaxt, rrh, rminrh, rwindspd, rsunpct
80  real(kind=T_SGL) :: rmonthavg,rmonthindex
81  character (len=256) :: sbuf
82  character (len=3) :: smonthname
83  logical (kind=T_LOGICAL) :: lmonthend
84  logical (kind=T_LOGICAL) :: lopened
85  integer (kind=T_INT) :: ilu
86  integer (kind=T_INT),dimension(12) :: imonthlycount
87  real (kind=T_INT),dimension(12) :: rmonthlysum
88  ! [ LOCAL CONSTANTS ]
89  real (kind=T_SGL),parameter :: rexp=1.514_t_sgl
90  real (kind=T_SGL),dimension(4),parameter :: rapoly = &
91  (/ 6.75e-07_t_sgl, -7.71e-05_t_sgl, 1.792e-02_t_sgl, 0.49239e+00_t_sgl /)
92 
93  write(unit=lu_log,fmt=*)"Initializing Thornthwaite-Mather ET model with annual data ", trim(sfilename)
94 
95  open ( lu_temp, file=trim(sfilename), iostat=istat )
96  call assert( istat == 0, "Could not open time series file " // trim(sfilename), &
97  trim(__file__),__line__)
98 
99  ! Now, compute the monthly-average temperatures for all months
100  imonthlycount = 0
101  rmonthlysum = rzero
102 
103  do
104  read ( unit=lu_temp, fmt="(a256)", iostat=istat ) sbuf
105  if ( istat<0 ) exit ! Here on EOF
106  call assert( istat == 0, "Cannot read record from time-series file" )
107  if ( sbuf(1:1) == '#' ) cycle ! Check comments
108  call cleanupcsv( sbuf )
109 
110  read ( unit=sbuf, fmt=*, iostat=istat ) imonth, iday, iyear, ravgt, &
111  rprecip, rrh, rmaxt, rmint, &
112  rwindspd, rminrh, rsunpct
113 
114  if (istat/=0) then
115  write(unit=lu_log,fmt=*)" Skipping: ",trim(sbuf)
116  cycle
117  end if
118 
119  call lookupmonth(imonth, iday, iyear,idayofyear, &
120  smonthname,lmonthend)
121 
122  rmonthlysum(imonth) = rmonthlysum(imonth) + ravgt
123  imonthlycount(imonth) = imonthlycount(imonth) + 1
124  end do
125 
126  ! Now, compute monthly temperature indices (leave out the no-data months)
127  ! And sum to get the annual index
128 
129  rannualindex = rzero
130  do i=1,12
131  if ( imonthlycount(i) > 0 ) then
132  rmonthavg = rmonthlysum(i)/imonthlycount(i)
133  if ( rmonthavg > rfreezing ) then
134  rmonthindex = (( rmonthavg - rfreezing )/9.0_t_sgl ) ** rexp
135  rannualindex = rannualindex + rmonthindex
136  end if
137  end if
138  end do
139 
140  rexponenta = polynomial( rannualindex, rapoly )
141 
142  close (unit=lu_temp)
143 
144  write(unit=lu_log,fmt=*) " Thornthwaite-Mather annual heat index = ",rannualindex
145  write(unit=lu_log,fmt=*) " Thornthwaite-Mather exponent 'a' = ",rexponenta
146 
147 end subroutine et_tm_initialize

149 subroutine et_tm_computeet( pGrd, pConfig, iDayOfYear, rRH, rMinRH, rWindSpd, rSunPct )
150  !! Computes the potential ET for each cell, based on the meteorological
151  !! data given. Stores cell-by-cell PET values in the model grid.
152  !! Note: for the T-M model, it's constant scross the grid
153  !!
154  ! [ ARGUMENTS ]
155  type ( t_general_grid ),pointer :: pgrd
156  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
157  ! model options, flags, and other settings
158  integer (kind=T_INT),intent(in) :: idayofyear
159  real (kind=T_SGL),intent(in) :: rrh,rminrh,rwindspd,rsunpct
160  real (kind=T_SGL) :: rdecl,rlcf,rpotet, rtempet
161 
162  ! [ locals ]
163  integer (kind=T_SGL) :: iavgt, icol, irow
164  real (kind=T_SGL),dimension(3),parameter :: rhightpoly = &
165  (/ -5.25625072726565d-03, 1.04170341298537d+00, - 44.3259754866234d+00 /)
166 
167 
168  rdecl = 0.4093_t_sgl * sin((dptwopi * idayofyear/365.0_t_sgl) - 1.405_t_sgl )
169 
170  rlcf = (24.0_t_sgl / dppi) * acos(-rtanlatitude * tan(rdecl)) / 12.0_t_sgl
171 
172  do irow=1,pgrd%iNY
173  do icol=1,pgrd%iNX ! last subscript in a Fortran array should be the slowest changing
174 
175  if (pgrd%Cells(icol,irow)%rTAvg <= rfreezing ) then
176  rpotet = rzero
177  else if ( pgrd%Cells(icol,irow)%rTAvg <= 79.7_t_sgl ) then
178 ! rPotET = ( 50.0_T_SGL*(rAvgT-rFREEZING) / (9.0_T_SGL * rAnnualIndex) ) ** rExponentA * &
179 ! rLCF * 0.63_T_SGL / 30.0_T_SGL
180  rpotet = ((0.63_t_sgl * rlcf) * ((50.0_t_sgl * &
181  (pgrd%Cells(icol,irow)%rTAvg - rfreezing) &
182  / (9_t_sgl * rannualindex)) ** rexponenta)) / 30_t_sgl
183  else
184  rpotet = rlcf * polynomial( pgrd%Cells(icol,irow)%rTAvg, rhightpoly ) / 30.0_t_sgl
185  end if
186 
187  pgrd%Cells(icol,irow)%rSM_PotentialET = rpotet
188 
189  end do
190 
191  end do
192 
193 ! write(UNIT=LU_LOG,FMT=*) "=========POTET CALCULATION==========="
194 ! write(UNIT=LU_STD_OUT,FMT="(A)") &
195 ! " min mean max"
196 ! call stats_WriteMinMeanMax(LU_STD_OUT,"Potet ET (in)" , pGrd%Cells(:,:)%rSM_PotentialET )
197 !
198 ! write(UNIT=LU_LOG,FMT=*) "=========POTET CALCULATION==========="
199 
200  return
201 
202 end subroutine et_tm_computeet
