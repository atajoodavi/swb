00888 !  Searches in the y-direction, given the x value and the table value
00889 !  z, and returns the value of y. Assumes that the row and
00890 !  column spacing are constant. The search begins at the _top_ of the

00892 !  Parameter 'rNoData' is the missing value code for the grid.
00893 
00894 ! INPUTS
00895 !  pGrd - Pointer to a data grid
00896 !  rXval - X value corresponding to one or more rows of the table.
00897 !  rZval - Table value to scan for.
00898 !  rNoData - Real value assigned to "No Data" cells.
00899 !
00900 ! OUTPUTS
00901 !  rYval - Returned value associated with the input x and z values.
00902 !
00903 ! NOTES
00904 !  The grid lookup functions were designed specifically to work with the
00905 !  Thornthwaite-Mather soil moisture retention table. The functions will
00906 !  work with any data that are similarly arranged.
00907 !
00908 !  If X=1.00 and Z=0.90, for example, and the following table were queried,
00909 !  the subroutine will return: rValue=0.1
00910 !
00911 !                 X=Max SM Capacity
00912 !
00913 !                0.50  1.00  1.50
00914 !               _________________
00915 !          0.0 | 0.50  1.00  1.50
00916 !  Y=APWL  0.1 | 0.45  0.90  1.40
00917 !          0.2 | 0.40  0.80  1.30
00918 !
00919 ! SOURCE
00920 
00921 function grid_SearchColumn(pGrd,rXval,rZval,rNoData) result ( rValue )
00922   !! Searches in the y-direction, given the x value 'xval', for the value
00923   !! 'zval', and returns the value of y in 'ry'. Assumes that the row and
00924   !! column spacing are constant. The search begins at the _top_ of the
00925   !! table (y=pGrd%rY1). Applicable only to T_SGL_GRID grids.
00926   !! Parameter 'rmv' is the missing value code for the grid.
00927   ! [ ARGUMENTS ]
00928   type ( T_GENERAL_GRID ),pointer :: pGrd
00929   real (kind=T_SGL),intent(in) :: rXval,rZval,rNoData
00930   ! [ RETURN VALUE ]
00931   real (kind=T_SGL) :: rValue
00932   ! [ LOCALS ]
00933   integer (kind=T_INT) :: ib,ia,istat,j
00934   real (kind=T_SGL) :: u,v,frac,rprev
00935   real (kind=T_SGL),dimension(:),allocatable :: rcol
00936   character (len=128) :: buf
00937 
00938   allocate ( rcol(pGrd%iNY), stat=istat )
00939   call Assert( LOGICAL(istat==0,kind=T_LOGICAL), 
00940      "Couldn't allocate temporary array" )
00941 
00942   call grid_LookupColumn(pGrd,rXval,ib,ia,u)
00943 !  write(UNIT=LU_LOG,FMT=*)'lookup ',rXval,ib,ia
00944 
00945 
00946   write (unit=buf,fmt=*) "Requested X value ", rXval, " out of range"
00947   call Assert (LOGICAL(ib>0 .and. ia>0,kind=T_LOGICAL), buf )
00948   rcol = u*pGrd%rData(:,ia) + (rONE-u)*pGrd%rData(:,ib)
00949   ! Fix missing values
00950   do j=1,pGrd%iNY
00951     if ( pGrd%rData(j,ia) == rNoData .and. pGrd%rData(j,ib) == rNoData ) then
00952       rcol(j) = rNoData
00953     else if ( pGrd%rData(j,ib) == rNoData .and. u>0.9_T_SGL ) then
00954       rcol(j) = pGrd%rData(j,ia)
00955     else if ( pGrd%rData(j,ia) == rNoData .and. u<0.1_T_SGL ) then
00956       rcol(j) = pGrd%rData(j,ib)
00957     else if ( pGrd%rData(j,ia) == rNoData .or. pGrd%rData(j,ib) == rNoData ) then
00958       rcol(j) = rNoData
00959     end if
00960   end do
00961 
00962   ! Search for the specified value, assuming that the data vary monotonically
00963   ! Skip MVs
00964   rprev = rNoData
00965   rValue = rNoData
00966   do j=1,pGrd%iNY
00967     if ( rcol(j) == rNoData ) then
00968       if ( rprev == rNoData ) then
00969         ! keep skipping missing values...
00970         continue
00971       else
00972         ! end of the line -- we didn't find the value, so return the limit
00973         v = real(j-1) / real(pGrd%iNY-1)
00974         rValue = (rONE-v)*pGrd%rY0 + v*pGrd%rY1
00975         exit
00976       endif
00977     else
00978       if ( rprev == rNoData ) then
00979         rprev = rcol(j)
00980       else
00981         if ( sign(rONE,rcol(j)-rZval) /= sign(rONE,rprev-rZval) ) then
00982           ! Found it!
00983           frac = (rZval-rcol(j-1)) / (rcol(j)-rcol(j-1))
00984           ! Note: it's 'j-2' in the next expr to account for one-based indexing
00985           v = ( real(j-2)+frac ) / real(pGrd%iNY-1)
00986           rValue = v*pGrd%rY0 + (rONE-v)*pGrd%rY1
00987           exit
00988         else if ( rZval < rprev .and. rcol(j) > rprev ) then
00989           ! Doesn't exist, choose the limit
00990           v = real(j-2) / real(pGrd%iNY-1)
00991           rValue = v*pGrd%rY0 + (rONE-v)*pGrd%rY1
00992           exit
00993         else if ( rZval > rprev .and. rcol(j) < rprev ) then
00994           v = real(j-2) / real(pGrd%iNY-1)
00995           rValue = v*pGrd%rY0 + (rONE-v)*pGrd%rY1
00996           exit
00997         else if ( j == pGrd%iNY ) then
00998           v = rONE
00999           rValue = v*pGrd%rY0 + (rONE-v)*pGrd%rY1
01000           exit
01001         end if
01002       end if
01003     end if
01004   end do
01005 
01006   deallocate ( rcol )
01007 
01008   return
01009 end function grid_SearchColumn
