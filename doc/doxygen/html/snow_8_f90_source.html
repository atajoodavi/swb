75 subroutine snow_energy_balance(rTMin, rTMax, rTAvg, rRs, rRso, rAlbedo, &
76  rsnowcover, rnetrainfall, rsnowtemperature, rmeltamount, i, j)
77 
78  real (kind=T_SGL):: rtmin
79  real (kind=T_SGL) :: rtmax
80  real (kind=T_SGL) :: rtavg
81  real (kind=T_DBL) :: rrs
82  real (kind=T_DBL) :: rrso
83  real (kind=T_SGL) :: ralbedo
84  real (kind=T_SGL) :: rsnowcover
85  real (kind=T_SGL) :: rnetrainfall
86  real (kind=T_SGL) :: rsnowtemperature
87  real (kind=T_SGL) :: rmeltamount
88  real (kind=T_SGL) :: rmeltpotential
89  integer (kind=T_INT) :: i,j
90 
91  ! [ LOCALS ]
92  real (kind=T_SGL) :: rsn ! net shortwave radiation
93  real (kind=T_SGL) :: rln ! net longwave radiation
94  real (kind=T_SGL) :: rh ! sensible heat exchange
95  real (kind=T_SGL) :: re ! convective heat exchange
96  real (kind=T_SGL), parameter :: rg = 173. ! ground heat, kj per square meter
97  real (kind=T_SGL) :: rp ! precipitation heat exchange
98  real(kind=T_SGL) :: rswe
99  real (kind=T_DBL) :: rdeltat
100  real (kind=T_DBL) :: rdeltaswe
101  real (kind=T_SGL) :: rsnowtemp_degc
102  real (kind=T_SGL) :: rtmin_c
103  real (kind=T_DBL) :: rnetenergy
104  real (kind=T_DBL) :: rt_riseenergy
105  real (kind=T_DBL) :: rmeltenergy
106  real (kind=T_DBL), parameter :: rcsnow = 2.1 ! kJ per kg per deg C
107  ! snow heat capacity
108 
109  real (kind=T_DBL), parameter :: rlambda = 3.35e+05 ! kJ per cubic meter per deg C
110  ! latent heat of fusion
111 
112  ! Snowcover is in INCHES SWE; we need METERS SWE
113 
114  rswe = rsnowcover / 12.0 *.3048
115  rtmin_c = ftoc(rtmin)
116  rsnowtemp_degc = ftoc(rsnowtemperature)
117 
118  ! all component energy values must be in kJ per square meter per day
119  ! radiation components must be multiplied by 1000 since they are returned
120  ! in MJ per square meters per day
121  rsn = net_shortwave_radiation_rns(rrs, real(ralbedo, kind=t_dbl)) * 1000_t_dbl
122  rln = net_longwave_radiation_rnl(rtmin, rtmax, rrs, rrso) * 1000_t_dbl
123  rh = sensible_heat_exchange_h(rsnowtemperature, rtavg)
124  re = convective_heat_exchange_e(rsnowtemperature, rtavg)
125  rp = precipitation_heat_p(rnetrainfall, rtavg)
126 
127  if(i==3 .and. j==3) then
128  write(*,fmt="(' rSnowTemperature (input): ',F14.4)") rsnowtemperature
129  write(*,fmt="(' rSWE (input) : ',F14.4)") rswe
130  end if
131 
132  if(rswe > 0) then ! we have snow and should do the calculations
133 
134  rnetenergy = ( rsn - rln + rh + re + rg + rp )
135  rdeltat = 0.01 * rnetenergy / (rswe * rcsnow * 1000.)
136 
137  if(i==3 .and. j==3) then
138  write(*,fmt="(' rNetEnergy : ',F14.4)") rnetenergy
139  write(*,fmt="(' rDeltaT : ',F14.4)") rdeltat
140  end if
141 
142 
143  if(rnetenergy < 0.) then ! insufficient energy is available for snowmelt
144  ! snowpack temperature decreases
145 
146  if(i==3 .and. j==3) print *, ' snowpack temp DECREASES'
147  rsnowtemp_degc = rsnowtemp_degc + rdeltat
148  if( rsnowtemp_degc > 0.) rsnowtemp_degc = 0.
149  if(rsnowtemp_degc < rtmin_c) rsnowtemp_degc = rtmin_c
150  rmeltamount = 0.
151  ! now recalculate rDeltaT after constraints applied to rSnowTemp_degC
152  rdeltat = ftoc(rsnowtemperature) - rsnowtemp_degc
153  else ! excess energy added to snowpack; possible snowmelt
154 
155  if(rsnowtemp_degc + rdeltat > 0.) then ! more energy added to snowpack than
156  ! it can absorb; melting occurs
157  if(i==3 .and. j==3) print *, ' snowpack temp DECREASES; MELTING OCCURS'
158 
159  rt_riseenergy = (rswe * rcsnow * 1000. ) * abs( rsnowtemp_degc )
160  rsnowtemp_degc = 0.
161  rmeltenergy = rnetenergy - rt_riseenergy
162  rmeltpotential = rmeltenergy / (333.3 * 1000.)
163  rdeltaswe = min(rswe,rmeltpotential)
164  rmeltamount = abs(rdeltaswe) / 0.3048 * 12.
165  rsnowcover = max(rsnowcover - rmeltamount,rzero)
166 
167  else ! sufficient energy to raise snowpack temperature, but no snowmelt occurs
168  if(i==3 .and. j==3) print *, ' snowpack temp INCREASES'
169  rmeltamount = 0.
170  rsnowtemp_degc = rsnowtemp_degc + rdeltat
171 
172  end if
173  end if
174  else
175  rmeltamount = 0.
176  rsnowcover = 0.
177  end if
178 
179  rsnowtemperature = ctof(rsnowtemp_degc)
180 
181 end subroutine snow_energy_balance

189 ! Estimates snow albedo given number of days since last snowfall

192 ! rAlbedoInit - initial albedo when snow first falls
193 ! iNumDaysLastSnow - number of days since the last snowfall took place

195 !
196 ! OUTPUTS
197 ! rAlbedo - current albedo of snow surface
198 !
199 ! NOTES
200 !
201 ! Implemented as equation 13-15, Kustas et al, 1994
202 !
203 ! Reference:
204 ! Kustas, W.P., Rango, A. and Uijlenhoet, R., A simple energy
205 ! budget algorithm for the snowmelt runoff model:
206 ! Water Resources Research, v. 30, no. 5.
207 !
208 ! SOURCE
209 
210 function snow_albedo(rAlbedoInit, iNumDaysLastSnow, rZenithAngle) result(rAlbedo)
211 
212  real (kind=T_SGL), intent(in) :: ralbedoinit
213  integer (kind=T_INT), intent(in) :: inumdayslastsnow
214  real (kind=T_SGL), intent(in) :: rzenithangle
215 
216  ! [ LOCALS ]
217  real (kind=T_SGL) :: ralbedo
218  real (kind=T_SGL) :: rsnowgrainsize
219  real (kind=T_SGL), parameter :: rsnowgrainsizebeforemelting = 0.25 ! millimeters
220 
221  rsnowgrainsize = (rsnowgrainsizebeforemelting * rsnowgrainsizebeforemelting &
222  * rsnowgrainsizebeforemelting &

224 
225  ralbedo = ralbedoinit - (0.083_t_sgl + 0.23_t_sgl * sqrt(rsnowgrainsize)) &
