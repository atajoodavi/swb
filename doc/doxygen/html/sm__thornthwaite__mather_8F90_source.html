70 subroutine sm_thornthwaite_mather_initialize ( pGrd, pConfig )
71  !! Preconfigures soil moisture for the model grid 'pGrd'.
72  ! [ ARGUMENTS ]
73  type ( t_general_grid ),pointer :: pgrd
74  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
75  ! model options, flags, and other settings
76  ! [ LOCALS ]
77  integer (kind=c_int) :: irow,icol
78  type ( t_cell ),pointer :: cel
79 
80  ! Initialize the accumulated water loss for each cell according to the
81  ! initial soil moisture.
82 
83  do irow=1,pgrd%iNY
84  do icol=1,pgrd%iNX
85 
86  cel => pgrd%Cells(icol,irow)
87 
88  if (pgrd%iMask(icol, irow) == iinactive_cell) cycle
89 
90  if ( cel%rSoilWaterCap > rnear_zero ) then
91 
92  if (pconfig%iConfigureSM == config_sm_tm_lookup_table) then
93 
94  !! Constrain rSoilWaterCap to limits of Thornthwaite-Mather tables
95 
96  if ( cel%rSoilWaterCap < 0.5_c_float ) then
97  write(unit=lu_log,fmt="(a,f10.3,a,i4,',',i4,a)") &
98  'Soil water capacity', cel%rSoilWaterCap,' for cell ',irow,icol, &
99  ' out of range; adjusted to 0.5'
100  flush(unit=lu_log)
101  cel%rSoilWaterCap = 0.51_c_float
102  else if ( cel%rSoilWaterCap > 17.5_c_float ) then
103  write(unit=lu_log,fmt="(a,f10.3,a,i4,',',i4,a)") &
104  'Soil water capacity', cel%rSoilWaterCap,' for cell ',irow,icol, &
105  ' out of range; adjusted to 17.5'
106  flush(unit=lu_log)
107  cel%rSoilWaterCap = 17.49_c_float
108  end if
109 
110  !! convert input soil moisture (as percent of soil water capacity)
111  !! TO soil moisture in inches
112  cel%rSoilMoisture = (cel%rSoilMoisturePct/rhundred) * cel%rSoilWaterCap
113 
114  ! calculate APWL from T-M table
115  cel%rSM_AccumPotentWatLoss = grid_searchcolumn(pgrd=gwlt, &
116  rxval=cel%rSoilWaterCap, &
117  rzval=cel%rSoilMoisture, &
118  rnodata=-rone)
119 
120  elseif ( pconfig%iConfigureSM == config_sm_tm_equations ) then
121 
122  !! convert input soil moisture (as percent of soil water capacity)
123  !! TO soil moisture in inches
124  cel%rSoilMoisture = (cel%rSoilMoisturePct/rhundred) * cel%rSoilWaterCap
125 
126  ! calculate APWL from equation
127  cel%rSM_AccumPotentWatLoss = &
128  sm_thornthwaite_mather_apwl(cel%rSoilWaterCap,real(cel%rSoilMoisture, kind=c_double) )
129 
130  endif
131 
132  end if
133 
134  end do
135 
136  end do
137 
138 #ifdef DEBUG_PRINT
139  call grid_writearcgrid("initial_APWL.asc",dpzero,dpone,dpzero, &
140  dpone,pgrd%Cells(:,:)%rSM_AccumPotentWatLoss)
141 #endif
142 
143  !! initialize Thorntwaite-Mather soil moisture accounting variables
144 
145  pgrd%Cells%rReferenceET0 = rzero
146 
147 end subroutine sm_thornthwaite_mather_initialize

152 function sm_thornthwaite_mather_soil_storage(rSWC, rAPWL) result(dpValue)
153 
154  real (kind=c_float), intent(in) :: rswc ! max soil-water capacity (inches)
155  real (kind=c_float), intent(in) :: rapwl ! accum pot. water loss (inches)
156 
157  real (kind=c_double) :: dpvalue
158 
159  ! equation as implemented in R;
160  ! sm.df$y = maximum soil-water capacity
161  ! sm.df$x = APWL
162  ! 10^(log10(sm.df$y) - (s.opt[[1]]*sm.df$y^e.opt[[1]]) * sm.df$x)
163 
164  dpvalue = dpzero
165 
166  if(rswc > rzero ) &
167 
168  dpvalue = 10_c_double**( log10(REAL(rswc,kind=c_double)) - &
169  ( abs(REAL(rapwl,kind=c_double)) * rtm_slope_term &
170  * real(rswc, kind=c_double)**rtm_exp_term ) )
171 
172 
173 end function sm_thornthwaite_mather_soil_storage

177 function sm_thornthwaite_mather_apwl(rSWC, dpSoilStorage) result(dpValue)
178 
179  real (kind=c_float), intent(in) :: rswc ! max soil-water capacity (inches)
180  real (kind=c_double), intent(in) :: dpsoilstorage ! curr soil storage (inches)
181 
182  real (kind=c_double) :: dpvalue
183 
184  ! equation as implemented in R;
185  ! sm.df$y = maximum soil-water capacity
186  ! sm.df$x = APWL
187  ! (log10(sm.df$y) - log10(sm.df$pred)) / (s.opt[[1]] * sm.df$y^e.opt[[1]])
188 
189  dpvalue = dpzero
190 
191  if(rswc > rzero .and. dpsoilstorage > dpzero) &
192 
193  dpvalue = -( log10(REAL(rswc,kind=c_double)) - log10(dpsoilstorage)) / &
194  ( rtm_slope_term * REAL(rswc,kind=c_double)**rtm_exp_term )
195
