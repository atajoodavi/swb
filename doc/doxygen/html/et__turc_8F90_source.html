83 subroutine et_turc_computeet( pGrd, iDayOfYear, rRH, rSunPct )
84  !! Computes the potential ET for each cell, based on the meteorological
85  !! data given. Stores cell-by-cell PET values in the model grid.
86  !! Note: for the T-M model, it's constant scross the grid
87  !!
88  ! [ ARGUMENTS ]
89  type ( t_general_grid ),pointer :: pgrd
90  integer (kind=c_int),intent(in) :: idayofyear
91  real (kind=c_float),intent(in) :: rrh,rsunpct
92  ! [ LOCALS ]
93  real (kind=c_float) :: rpotet,rso,rdelta,romega_s,rd_r,rs0,rsn,rt
94  integer (kind=c_int) :: icol,irow
95  ! [ CONSTANTS ]
96  real (kind=c_float),parameter :: unit_conv = 0.313_c_float / 25.4_c_float
97 
98  call assert( LOGICAL(rSunPct>=rZERO,kind=c_bool), "Missing data for percent sunshine" )
99  call assert( LOGICAL(rRH>=rZERO,kind=c_bool), "Missing data for relative humidity" )
100 
101  rd_r = rone + 0.033_c_float * cos( dptwopi * idayofyear / 365.0_c_float )
102  rdelta = 0.4093_c_float * sin( (dptwopi * idayofyear / 365.0_c_float) - 1.405_c_float )
103  romega_s = acos( -tan(rlatitude) * tan(rdelta) )
104  rso = 2.44722_c_float * 15.392_c_float * rd_r * ( romega_s * sin(rlatitude) * sin(rdelta) + &
105  sin(romega_s) * cos(rlatitude) * cos(rdelta) )
106  rsn = rso * ( rone-ralbedo ) * ( ras + rbs * rsunpct / rhundred )
107 
108  do irow=1,pgrd%iNY
109  do icol=1,pgrd%iNX ! last subscript in a Fortran array should be the slowest changing
110 
111  if ( pgrd%Cells(icol,irow)%rTAvg <= rfreezing ) then
112  rpotet = rzero
113  else
114  rt = ftoc(pgrd%Cells(icol,irow)%rTAvg)
115  rpotet = unit_conv * rt * ( rsn+2.1_c_float ) / ( rt + 15.0_c_float )
116  if ( rrh < 50.0_c_float ) then
117  rpotet = rpotet * ( rone + (50.0_c_float - rrh) / 70.0_c_float )
118  end if
119  end if
120 
121  pgrd%Cells(icol,irow)%rReferenceET0 = rpotet
122 
123  end do
124 
125  end do
126 
127  return
128 
129 end subroutine et_turc_computeet
