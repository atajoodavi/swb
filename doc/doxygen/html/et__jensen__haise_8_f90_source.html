11 ! et_jensen_haise.f95 - Evapotranspiration calculation using the

14 ! SYNOPSIS
15 ! This module calculates evapotranspiration using the Jensen-Haise
16 ! method.
17 !
18 ! NOTES
19 ! Original method is documented in:
20 !
21 ! Jensen, M.E., H.R. Haise. 1963. Estimating evapotranspiration from solar

40 subroutine et_jh_configure( sRecord )
41  !! Configures the module, using the command line in 'sRecord'
42  ! [ ARGUMENTS ]
43  character (len=*),intent(inout) :: srecord
44  ! [ LOCALS ]
45  character (len=256) :: soption
46  integer (kind=T_INT) :: istat
47 
48  write(unit=lu_log,fmt=*) "Configuring Jensen-Haise PET model"
49 
50  call chomp( srecord,soption )
51  read ( unit=soption, fmt=*, iostat=istat ) rlatitude
52  call assert( istat == 0, "Could not read the latitude" )
53  rlatitude = dptwopi * rlatitude / 360.0_t_sgl
54 
55  call chomp( srecord,soption )

83 subroutine et_jh_computeet( pGrd, iDayOfYear, rRH, &
84  rminrh, rwindspd, rsunpct )
85  !! Computes the potential ET for each cell, based on the meteorological
86  !! data given. Stores cell-by-cell PET values in the model grid.
87  !! Note: for the T-M model, it's constant scross the grid
88  !!
89  ! [ ARGUMENTS ]
90  type ( t_general_grid ),pointer :: pgrd
91  integer (kind=T_INT),intent(in) :: idayofyear
92  real (kind=T_SGL),intent(in) :: rrh,rminrh,rwindspd,rsunpct
93  ! [ LOCALS ]
94  real (kind=T_SGL) :: rso,rdelta,romega_s,rd_r,rs0,rsn,rt
95  integer (kind=T_INT) :: icol, irow
96  ! [ CONSTANTS ]
97  real (kind=T_SGL),parameter :: unit_conv = 0.41_t_sgl / 25.4_t_sgl
98 
99  call assert( LOGICAL(rSunPct>=rZERO, kind=T_LOGICAL),"Missing data for percent sunshine" )
100  call assert( LOGICAL(rRH>=rZERO, kind=T_LOGICAL),"Missing data for relative humidity" )
101 
102  rd_r = rone + 0.033_t_sgl * cos( dptwopi * idayofyear / 365.0_t_sgl )
103  rdelta = 0.4093_t_sgl * sin( (dptwopi * idayofyear / 365.0_t_sgl) - 1.405_t_sgl )
104  romega_s = acos( -tan(rlatitude) * tan(rdelta) )
105  rso = 2.44722_t_sgl * 15.392_t_sgl * rd_r * ( romega_s * sin(rlatitude) * sin(rdelta) + &
106  sin(romega_s) * cos(rlatitude) * cos(rdelta) )
107  rsn = rso * ( rone-ralbedo ) * ( ras + rbs * rsunpct / rhundred )
108 
109  do irow=1,pgrd%iNY
110  do icol=1,pgrd%iNX ! last subscript in a Fortran array should be the slowest changing
111 
112  if ( pgrd%Cells(icol,irow)%rTAvg <= rfreezing ) then
113  pgrd%Cells(icol,irow)%rSM_PotentialET = rzero
114  else
115  rt = ftoc(pgrd%Cells(icol,irow)%rTAvg)
116  pgrd%Cells(icol,irow)%rSM_PotentialET = unit_conv * ( 0.025_t_sgl * rt + 0.078_t_sgl ) * rsn
117  end if
118 
119  end do
120 
121  end do
122 
123  return
124 
125 end subroutine et_jh_computeet
