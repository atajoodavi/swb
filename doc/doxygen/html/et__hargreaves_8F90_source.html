79  !! Computes the potential ET for each cell, based on TMIN and TMAX.
80  !! Stores cell-by-cell PET values in the model grid.
81 
82  !!
83  ! [ ARGUMENTS ]
84  type ( t_general_grid ),pointer :: pgrd
85  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
86  ! model options, flags, and other settings

89  integer (kind=T_INT),intent(in) :: inumdaysinyear
90 
91  ! [ LOCALS ]
92  real (kind=T_SGL) :: rdelta,romega_s,rd_r, rra
93  real (kind=T_SGL) :: rlatitude
94  integer (kind=T_INT) :: icol, irow
95 
96 ! write(UNIT=LU_LOG,FMT=*) iDayOfYear, iNumDaysInYear
97 
98  rd_r =rel_earth_sun_dist(idayofyear,inumdaysinyear)
99  rdelta = solar_declination(idayofyear, inumdaysinyear)
100 
101  do irow=1,pgrd%iNY
102 
103  rlatitude = row_latitude(pconfig%rNorthernLatitude, &
104  pconfig%rSouthernLatitude, pgrd%iNY, irow)
105  romega_s = sunset_angle(rlatitude, rdelta)
106 
107  ! NOTE that the following equation returns extraterrestrial radiation in
108  ! MJ / m**2 / day. The Hargreaves equation requires extraterrestrial
109  ! radiation to be expressed in units of mm / day.
110  rra = extraterrestrial_radiation_ra(rlatitude,rdelta,romega_s,rd_r)
111 
112 ! write(UNIT=LU_LOG,FMT=*) "Row: ",iRow," Latitude: ",rLatitude, " N_Lat: ",pConfig%rNorthernLatitude, &
113 ! " S_Lat: ",pConfig%rSouthernLatitude, " Ra: ", rRa
114 
115  do icol=1,pgrd%iNX
116 
117  pgrd%Cells(icol,irow)%rSM_PotentialET = et0_hargreaves( &
118  pconfig, &
119  equivalent_evaporation(rra), &
120  pgrd%Cells(icol,irow)%rTMin, &
121  pgrd%Cells(icol,irow)%rTMax)

139 ! NAME
140 ! ET0_hargreaves - Calculates reference ET from air temperature data.
141 !
142 ! SYNOPSIS
143 ! Calculates reference ET given minimum and maximum air temperature.
144 !
145 ! INPUTS
146 ! r_mm - Value in millimeters.
147 !
148 ! OUTPUTS
149 ! r_in - Value in inches.
150 !
151 ! SOURCE
152 
153 function et0_hargreaves(pConfig, rRa, rTMinF, rTMaxF) result(rET_0)
154 
155  ! [ ARGUMENTS ]
156 
157  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
158  ! model options, flags, and other settings
159  real (kind=T_SGL),intent(in) :: rra
160  real (kind=T_SGL),intent(in) :: rtminf
161  real (kind=T_SGL),intent(in) :: rtmaxf
162 
163  ! [ RETURN VALUE ]
164  real (kind=T_SGL) :: ret_0
165 
166  ! [ LOCALS ]
167  real (kind=T_SGL) :: rtdelta
168  real (kind=T_SGL) :: rtavg
169 
170  rtavg = (rtminf + rtmaxf) / 2_t_sgl
171 
172  rtdelta = ftok(rtmaxf) - ftok(rtminf)
173 
174 ! rET_0 = MAX(rZERO, &
175 ! ( 0.0023_T_SGL &
176 ! * rRa &
177 ! * (FtoC(rTavg) + 17.8_T_SGL) &
178 ! * sqrt(rTDelta)) &
179 ! / rMM_PER_INCH)
180 
181  ret_0 = max(rzero, &
182  ( pconfig%rET_Slope &
183  * rra &
184  * (ftoc(rtavg) + pconfig%rET_Constant) &
185  * (rtdelta**pconfig%rET_Exponent)) &
186  / rmm_per_inch)
187 
188  return
189 
190 end function et0_hargreaves
