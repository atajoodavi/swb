55 function grid_create ( iNX, iNY, rX0, rY0, rX1, rY1, iDataType ) result ( pGrd )
56  !! Creates a new iNX-by-iNY T_GRID of data type iDataType, over the extent

78  call assert( istat == 0, &

283  ! ARGUMENTS
284  character (len=*), intent(in) :: sfilename ! Name of the grid input file

291  character (len=256) :: sargument ! Argument for keyword directives
292  character (len=8192) :: sbuf
293  integer (kind=T_INT) :: istat ! For "iostat="
294  integer (kind=T_INT) :: inx,iny ! Grid dimensions
295  integer (kind=T_INT) :: ihdrrecs ! Number of records in header
296  integer (kind=T_INT) :: icol,irow,k ! Loop indices for grid reading
297  real (kind=T_DBL) :: rx0,rx1 ! Limits in X
298  real (kind=T_DBL) :: ry0,ry1 ! Limits in Y
299  real (kind=T_SGL) :: rcellsize ! Cell size
300  integer (kind=T_INT) :: icount,icumlcount
301  logical (kind=T_LOGICAL) :: lxllcenter, lyllcenter ! Flags XLLCENTER / XLLCORNER
302  logical (kind=T_LOGICAL) :: lfileexists
303 
304  ! Pre-scan for the number of header records and read the header
305  inquire(file=trim(sfilename), exist=lfileexists)
306  call assert( lfileexists, "The Arc ASCII grid file "//dquote(sfilename)// &
307  " could not be found.",trim(__file__),__line__)
308 
309  open ( lu_grid, iostat=istat, file=trim(sfilename) )
310  call assert( istat == 0, &
311  "Could not open input file " // trim(sfilename) )
312 
313  ihdrrecs = 0
314  lxllcenter = lfalse
315  lyllcenter = lfalse
316  do
317  read ( unit=lu_grid, fmt="(a256)", iostat=istat ) sinputrecord
318  call assert( istat == 0, &
319  "Could not read input record - file:"//trim(sfilename) )
320  call chomp( sinputrecord, sdirective )
321  call uppercase( sdirective )
322  call chomp( sinputrecord, sargument )
323  if ( sdirective == "NCOLS" ) then
324  read ( unit=sargument, fmt=*, iostat=istat ) inx
325  call assert( istat == 0, "Could not read NCOLS" )
326  ihdrrecs = ihdrrecs + 1
327  else if ( sdirective == "NROWS" ) then
328  read ( unit=sargument, fmt=*, iostat=istat ) iny
329  call assert( istat == 0, "Could not read NROWS" )
330  ihdrrecs = ihdrrecs + 1
331  else if ( sdirective == "XLLCENTER" ) then
332  read ( unit=sargument, fmt=*, iostat=istat ) rx0
333  call assert( istat == 0, "Could not read XLLCENTER" )
334  lxllcenter = ltrue
335  ihdrrecs = ihdrrecs + 1
336  else if ( sdirective == "YLLCENTER" ) then
337  read ( unit=sargument, fmt=*, iostat=istat ) ry0
338  call assert( istat == 0, "Could not read YLLCENTER" )
339  lxllcenter = ltrue
340  ihdrrecs = ihdrrecs + 1
341  else if ( sdirective == "XLLCORNER" ) then
342  read ( unit=sargument, fmt=*, iostat=istat ) rx0
343  call assert( istat == 0, "Could not read XLLCORNER" )
344  ihdrrecs = ihdrrecs + 1
345  else if ( sdirective == "YLLCORNER" ) then
346  read ( unit=sargument, fmt=*, iostat=istat ) ry0
347  call assert( istat == 0, "Could not read YLLCORNER" )
348  ihdrrecs = ihdrrecs + 1
349  else if ( sdirective == "CELLSIZE" ) then
350  read ( unit=sargument, fmt=*, iostat=istat ) rcellsize
351  call assert( istat == 0, "Could not read CELLSIZE" )
352  ihdrrecs = ihdrrecs + 1
353  else if ( sdirective == "NODATA_VALUE" ) then
354  ihdrrecs = ihdrrecs + 1
355  else
356  ! Found the data -- construct the grid
357  if ( lxllcenter ) rx0 = rx0 - rhalf*rcellsize
358  if ( lyllcenter ) ry0 = ry0 - rhalf*rcellsize
359  rx1 = rx0 + inx*rcellsize
360  ry1 = ry0 + iny*rcellsize
361 
362  pgrd => grid_create( inx, iny, rx0, ry0, rx1, ry1, idatatype )
363 
364  pgrd%rGridCellSize = rcellsize
365  ! Go back to the top, skip the header...
366  rewind( unit=lu_grid, iostat=istat )
367  call assert( istat == 0, "Failed to rewind grid file" )
368  do icol=1,ihdrrecs
369  read ( unit=lu_grid, fmt="(a256)", iostat=istat ) sinputrecord
370  call assert( istat == 0, &
371  "Could not read input record - file: "//trim(sfilename))
372  end do
373  ! ... and read the data.
374  select case ( idatatype )
375  case ( t_int_grid )
376  do irow=1,pgrd%iNY
377  read ( unit=lu_grid, fmt=*, iostat=istat ) pgrd%iData(:,irow)
378  call assert( istat == 0, &
379  "Failed to read integer grid data - file: " &
380  //trim(sfilename)//" row num: "//trim(int2char(irow)), &
381  trim(__file__),__line__ )
382  end do
383  case ( t_sgl_grid )
384  do irow=1,pgrd%iNY
385  read ( unit=lu_grid, fmt=*, iostat=istat ) pgrd%rData(:,irow)
386  call assert( istat == 0, &
387  "Failed to read real grid data - file: " &
388  //trim(sfilename)//" row num: "//trim(int2char(irow)), &
389  trim(__file__),__line__ )
390  end do
391  case default
392  call assert( lfalse, &
393  "Internal error -- illegal ARC GRID data type", &
394  trim(__file__),__line__)
395  end select
396  exit
397  end if
398  end do
399 
400  if(idatatype==t_int_grid) then ! check for strange or illegal values
401  icumlcount = 0
402  write(lu_log,"(1x,'Summary of integer grid data values')")
403 
404  do k=0,maxval(pgrd%iData)
405  icount=count(pgrd%iData==k)
406  if(icount>0) then
407  icumlcount = icumlcount + icount
408  write(lu_log,fmt="(3x,i8,' grid cells have value: ',i8)") &
409  icount, k
410  end if
411  end do
412 
413  write(lu_log,fmt="(1x,a,t45,i12)") "Total number of grid cells: ", &
414  size(pgrd%iData)
415  write(lu_log,fmt="(1x,a,t45,i12)") &
416  "Total number of grid cells with value >= 0: ",icumlcount
417  flush(lu_log)
418  call assert(LOGICAL(size(pGrd%iData)==iCumlCount,kind=T_LOGICAL), &
419  "Illegal or missing values in integer grid file: "//trim(sfilename))
420 
421  end if
422 
423  close ( unit=lu_grid, iostat=istat )
424  call assert( istat == 0, "Failed to close grid file" )
425 
426  return

898 ! NAME
899 ! grid_WriteArcGrid - Reads an Surfer ASCII grid of a specified type.

901 ! SYNOPSIS
902 ! Write an ARC ASCII grid of specified data type
903 !
904 ! INPUTS
905 !
906 ! pGrd1 - Pointer to first grid
907 ! pGrd2 - Pointer to second grid
908 ! rTolerance - OPTIONAL - Allowable tolerance between the two grids

910 ! OUTPUTS
911 ! lConform - Logical, TRUE if the grids are compatible within the
912 ! specified tolerance, FALSE if otherwise
913 !
914 ! SOURCE
915 
916 function grid_conform ( pGrd1, pGrd2, rTolerance ) result ( lConform )
917  !! Returns .true. if the T_GRID objects conform (in terms of cell sizes and extents)
918  !! The optional argument rTolerance is the precision for checking the (floating-point)
919  !! extent coordinates (this defaults to rDEFAULT_TOLERANCE, below). The tolerance
920  !! is set to abs(rTolerance * ( rX1-rX1 ) )
921  ! ARGUMENTS
922  type (t_general_grid), pointer :: pgrd1,pgrd2

928  real (kind=T_SGL), parameter :: rdefault_tolerance = 1.0e-2_t_sgl
929 
930  if ( present( rtolerance ) ) then
931  rtol = rtolerance * ( pgrd1%rX1 - pgrd1%rX0 )
932  else
933  rtol = rdefault_tolerance * ( pgrd1%rX1 - pgrd1%rX0 )
934  end if
935 
936  if ( pgrd1%iNX /= pgrd2%iNX .or. &
937  pgrd1%iNY /= pgrd2%iNY .or. &
938  abs( pgrd1%rX0 - pgrd2%rX0 ) > rtol .or. &
939  abs( pgrd1%rY0 - pgrd2%rY0 ) > rtol .or. &
940  abs( pgrd1%rX1 - pgrd2%rX1 ) > rtol .or. &
941  abs( pgrd1%rY1 - pgrd2%rY1 ) > rtol ) then
942  lconform = lfalse
943  else
944  lconform = ltrue
945  end if
946 
947  return
948 end function grid_conform

955 !
956 ! SYNOPSIS
957 ! Finds the column position of the value rXval in the grid and returns:
958 ! iBefore = column before the value rYval
959 ! iAfter = column after the value rYval
960 ! rFrac = fraction of the distance between iBefore and iAfter
961 ! If iBefore or iAfter are outside the grid, they're set to -1
962 !
963 ! INPUTS
964 ! pGrd - Pointer to a data grid
965 ! rXval - Real value to be interpolated from the gridded data.
966 ! rTolerance - OPTIONAL - Allowable tolerance between the two grids
967 !
968 ! OUTPUTS
969 ! iBefore - column before the value rXval
970 ! iAfter - column after the value rXval
971 ! rFrac - fraction of the distance between iBefore and iAfter
972 !
973 ! If iBefore or iAfter are outside the grid, they're set to -1
974 !
975 ! NOTES
976 ! The grid lookup functions were designed specifically to work with the
977 ! Thornthwaite-Mather soil moisture retention table. The functions will
978 ! work with any data that are similarly arranged.
979 !
980 ! If X=0.65, for example, and the following table were queried, the subroutine
981 ! will return: iBefore=1, iAfter=2, rFrac=0.3
982 !
983 ! X=Max SM Capacity
984 !
985 ! 0.50 1.00 1.50
986 ! _________________
987 ! 0.0 | 0.50 1.00 1.50
988 ! Y=APWL 0.1 | 0.45 0.90 1.40
989 ! 0.2 | 0.40 0.80 1.30
990 !
991 ! SOURCE
992 
993 subroutine grid_lookupcolumn(pGrd,rXval,iBefore,iAfter,rFrac)
994 
995  ! [ ARGUMENTS ]
996  type ( t_general_grid ),pointer :: pgrd
997  real (kind=T_SGL),intent(in) :: rxval
998  integer (kind=T_INT),intent(out) :: ibefore,iafter
999  real (kind=T_SGL),intent(out) :: rfrac
1000  ! [ LOCALS ]
1001  real (kind=T_SGL) :: rrowposition
1002 
1003  rrowposition = (pgrd%iNX - 1) * (rxval - pgrd%rX0) / (pgrd%rX1 - pgrd%rX0)
1004  rfrac = rzero
1005  ibefore = floor(rrowposition) + 1
1006  iafter = ibefore + 1
1007  if ( ibefore > pgrd%iNX .or. ibefore < 1) then
1008  ibefore = -1
1009  iafter = -1
1010  else if ( iafter > pgrd%iNX ) then
1011  iafter = -1
1012  else
1013  rfrac = mod(rrowposition,rone)
1014  end if
1015 
1016  return
1017 end subroutine grid_lookupcolumn

1196 function grid_searchcolumn(pGrd,rXval,rZval,rNoData) result ( rValue )
1197  !! Searches in the y-direction, given the x value 'xval', for the value
1198  !! 'zval', and returns the value of y in 'ry'. Assumes that the row and
1199  !! column spacing are constant. The search begins at the _top_ of the
1200  !! table (y=pGrd%rY1). Applicable only to T_SGL_GRID grids.
1201  !! Parameter 'rmv' is the missing value code for the grid.
1202  ! [ ARGUMENTS ]
1203  type ( t_general_grid ),pointer :: pgrd
1204  real (kind=T_SGL),intent(in) :: rxval,rzval,rnodata
1205  ! [ RETURN VALUE ]
1206  real (kind=T_SGL) :: rvalue
1207  ! [ LOCALS ]
1208  integer (kind=T_INT) :: ib,ia,istat,irow
1209  real (kind=T_SGL) :: u,v,frac,rprev
1210  real (kind=T_SGL),dimension(:),allocatable :: rcol
1211  character (len=128) :: buf
1212 
1213  allocate ( rcol(pgrd%iNY), stat=istat )
1214  call assert( LOGICAL(istat==0,kind=T_LOGICAL), &
1215  "Couldn't allocate temporary array" )
1216 
1217  call grid_lookupcolumn(pgrd,rxval,ib,ia,u)
1218 ! write(UNIT=LU_LOG,FMT=*)'lookup ',rXval,ib,ia
1219 
1220 
1221  write (unit=buf,fmt=*) "Requested X value ", rxval, " out of range"
1222  call assert(LOGICAL(ib>0 .and. ia>0,kind=T_LOGICAL), buf )
1223 
1224  ! interpolate the column of values based on the columns of values
1225  ! that bracket the real value rXval
1226  rcol = u*pgrd%rData(:,ia) + (rone-u)*pgrd%rData(:,ib)
1227  ! Fix missing values
1228  do irow=1,pgrd%iNY
1229  if ( pgrd%rData(irow,ia) == rnodata .and. pgrd%rData(irow,ib) == rnodata ) then
1230  rcol(irow) = rnodata
1231  else if ( pgrd%rData(irow,ib) == rnodata .and. u>0.9_t_sgl ) then
1232  rcol(irow) = pgrd%rData(irow,ia)
1233  else if ( pgrd%rData(irow,ia) == rnodata .and. u<0.1_t_sgl ) then
1234  rcol(irow) = pgrd%rData(irow,ib)
1235  else if ( pgrd%rData(irow,ia) == rnodata .or. pgrd%rData(irow,ib) == rnodata ) then
1236  rcol(irow) = rnodata
1237  end if
1238  end do
1239 
1240  ! Search for the specified value, assuming that the data vary monotonically
1241  ! Skip MVs
1242  rprev = rnodata
1243  rvalue = rnodata
1244  do irow=1,pgrd%iNY
1245  if ( rcol(irow) == rnodata ) then
1246  if ( rprev == rnodata ) then
1247  ! keep skipping missing values...
1248  continue
1249  else
1250  ! end of the line -- we didn't find the value, so return the limit
1251  v = real(iRow-1) / real(pgrd%iny-1)
1252  rvalue = (rone-v)*pgrd%rY0 + v*pgrd%rY1
1253  exit
1254  endif
1255  else
1256  if ( rprev == rnodata ) then
1257  rprev = rcol(irow)
1258  else
1259  if ( sign(rone,rcol(irow)-rzval) /= sign(rone,rprev-rzval) ) then
1260  ! Found it!
1261  frac = (rzval-rcol(irow-1)) / (rcol(irow)-rcol(irow-1))
1262  ! Note: it's 'iRow-2' in the next expr to account for one-based indexing
1263  v = ( real(irow-2)+frac ) / real(pgrd%iny-1)
1264  rvalue = v*pgrd%rY0 + (rone-v)*pgrd%rY1
1265  exit
1266  else if ( rzval < rprev .and. rcol(irow) > rprev ) then
1267  ! Doesn't exist, choose the limit
1268  v = real(iRow-2) / real(pgrd%iny-1)
1269  rvalue = v*pgrd%rY0 + (rone-v)*pgrd%rY1
1270  exit
1271  else if ( rzval > rprev .and. rcol(irow) < rprev ) then
1272  v = real(iRow-2) / real(pgrd%iny-1)
1273  rvalue = v*pgrd%rY0 + (rone-v)*pgrd%rY1
1274  exit
1275  else if ( irow == pgrd%iNY ) then
1276  v = rone
1277  rvalue = v*pgrd%rY0 + (rone-v)*pgrd%rY1
1278  exit
1279  end if
1280  end if
1281  end if
1282  end do
1283 
1284  deallocate ( rcol )
1285 
1286  return
