103 function prob_runoff_enhancement(rCFGI, rLL, rUL) result(rPf)
104 
105  real (kind=T_SGL) :: rcfgi, rll, rul, rpf
106 
107  call assert(LOGICAL(rLL<=rUL,kind=T_LOGICAL), &
108  "Lower CFGI limit defining unfrozen ground must be <= upper CFGI limit")
109 
110  if(rcfgi <= rll) then
111  rpf = rzero
112  elseif(rcfgi >= rul) then
113  rpf = rone
114  else
115  rpf = (rcfgi - rll) / (rul - rll)
116  end if
117 
118  return

122 subroutine runoff_updatecurvenumber(pConfig, cel,iJulDay)
123  !! Updates the curve numbers for this iteration
124  ! [ ARGUMENTS ]
125  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that con
126  ! model options, flags, and other se
127  type (t_cell),pointer :: cel
128  integer (kind=T_INT),intent(in) :: ijulday
129  ! [ LOCALS ]
130  real (kind=T_SGL) :: rtotalinflow
131  real (kind=T_SGL) :: rtempcn
132  real (kind=T_SGL) :: rpf
133 
134  !! update curve numbers based on antecedent moisture conditions
135 ! real (kind=T_SGL),dimension(5),parameter :: rDRY_COEFS = (/ &
136 ! 1.44206581732462E-06_T_SGL, &
137 ! -2.54340415305462E-04_T_SGL, &
138 ! 2.07018739405394E-02_T_SGL, &
139 ! -7.67877072822852E-03_T_SGL, &
140 ! 2.09678222732103_T_SGL /)
141 ! real (kind=T_SGL),dimension(5),parameter :: rWET_COEFS = (/ &
142 ! -6.20352282661163E-07_T_SGL, &
143 ! 1.60650096926368E-04_T_SGL, &
144 ! -2.03362629006156E-02_T_SGL, &
145 ! 2.01054923513527_T_SGL, &
146 ! 3.65427885962651_T_SGL /)
147 
148  ! Here goes...
149  rtotalinflow = sum(cel%rNetInflowBuf)
150 
151  ! Correct the curve number...
152  if(cel%rCFGI>pconfig%rLL_CFGI &
153  .and. cel%rSoilWaterCap > rnear_zero) then
154 ! cel%rAdjCN = MIN(98_T_SGL,cel%rAdjCN + &
155 ! ((98_T_SGL-cel%rAdjCN) * cel%rSoilMoisture / cel%rSoilWaterCap))
156 
157  rpf = prob_runoff_enhancement(cel%rCFGI,pconfig%rLL_CFGI,pconfig%rUL_CFGI)
158 
159  ! use probability of runoff enhancement to calculate a weighted
160  ! average of curve number under Type II vs Type III antecedent
161  ! runoff conditions
162  cel%rAdjCN = cel%rBaseCN * (1-rpf) + &
163  (cel%rBaseCN / (0.427 + 0.00573 * cel%rBaseCN) * rpf)
164 
165  else if ( lf_model_growingseason(pconfig, ijulday) ) then
166 
167  if ( rtotalinflow < pconfig%rDRY_GROWING ) then ! AMC I
168 
169 
170 ! cel%rAdjCN = polynomial(cel%rBaseCN,rDRY_COEFS)
171 
172 ! The following comes from page 192, eq. 3.145 of "SCS Curve Number
173 ! Methodology"
174 
175  cel%rAdjCN = cel%rBaseCN / (2.281 - 0.01281 * cel%rBaseCN)
176 
177  else if ( rtotalinflow >= pconfig%rDRY_GROWING &
178  .and. rtotalinflow < pconfig%rWET_GROWING ) then ! AMC II
179 
180  cel%rAdjCN = real(cel%rbasecn)
181 
182  else ! AMC III
183 
184 ! cel%rAdjCN = polynomial(cel%rBaseCN,rWET_COEFS)
185  cel%rAdjCN = cel%rBaseCN / (0.427 + 0.00573 * cel%rBaseCN)
186  end if
187 
188  else ! dormant (non-growing) season
189 
190  if ( rtotalinflow < pconfig%rDRY_DORMANT ) then ! AMC I
191 
192 ! cel%rAdjCN = polynomial(cel%rBaseCN,rDRY_COEFS)
193  cel%rAdjCN = cel%rBaseCN / (2.281 - 0.01281 * cel%rBaseCN)
194 
195  else if ( rtotalinflow >= pconfig%rDRY_DORMANT &
196  .and. rtotalinflow < pconfig%rWET_DORMANT ) then ! AMC II
197 
198  cel%rAdjCN = real(cel%rbasecn)
199 
200  else ! AMC III
201 
202 ! cel%rAdjCN = polynomial(cel%rBaseCN,rWET_COEFS)
203  cel%rAdjCN = cel%rBaseCN / (0.427 + 0.00573 * cel%rBaseCN)
204 
205  end if
206 
207  end if
208 
209  ! ensure that whatever modification have been made to the curve number
210  ! remain within reasonable bounds
211  cel%rAdjCN = min(cel%rAdjCN,rhundred)
212  cel%rAdjCN = max(cel%rAdjCN,rzero)
213 
214  return
215 end subroutine runoff_updatecurvenumber

219 function runoff_cellrunoff_curvenumber(pConfig, cel, iJulDay) result(rOutFlow)
220  !! Calculates a single cell's runoff using curve numbers
221  type (t_model_configuration), pointer :: pconfig ! pointer to data structure that contains
222  ! model options, flags, and other settings
223  type (t_cell),pointer :: cel
224  integer (kind=T_INT),intent(in) :: ijulday
225  ! [ RETURN VALUE ]
226  real (kind=T_SGL) :: routflow
227  ! [ LOCALS ]
228  real (kind=T_SGL) :: rp
229  real (kind=T_SGL) :: rcn_05
230 
231  rp = cel%rNetPrecip &
232  + cel%rSnowMelt &
233 #ifdef IRRIGATION_MODULE
234  + cel%rIrrigationAmount &
235 #endif
236  + cel%rInFlow
237 
238  call runoff_updatecurvenumber(pconfig,cel,ijulday)
239 
240  cel%rSMax = (rthousand / cel%rAdjCN) - rten
241 
242  if(pconfig%iConfigureInitialAbstraction == &
243  config_sm_init_abstraction_tr55) then
244 
245  if ( rp > rpoint2*cel%rSMax ) then
246  routflow = ( rp - rpoint2*cel%rSMax )**2 / (rp + rpoint8*cel%rSMax)
247  else
248  routflow = rzero
249  end if
250 
251  else if(pconfig%iConfigureInitialAbstraction == &
252  config_sm_init_abstraction_hawkins) then
253 
254  ! Equation 9, Hawkins and others, 2002
255  rcn_05 = 100_t_sgl / &
256  ((1.879_t_sgl * ((100_t_sgl / cel%rAdjCN) - 1_t_sgl )**1.15_t_sgl) +1_t_sgl)
257 
258 
259  ! Equation 8, Hawkins and others, 2002
260  cel%rSMax = 1.33_t_sgl * ( cel%rSMax ) ** 1.15_t_sgl
261 
262  ! now consider runoff if Ia ~ 0.05S
263  if ( rp > 0.05_t_sgl*cel%rSMax ) then
264  routflow = ( rp - 0.05_t_sgl * cel%rSMax )**2 / (rp + 0.95_t_sgl*cel%rSMax)
265  else
266  routflow = rzero
267  end if
268 
269  else
270  call assert(lfalse, "Illegal initial abstraction method specified" )
271  end if
272 
273  return
274 
275 end function runoff_cellrunoff_curvenumber
